\subsection{Quantitative Trends}
\label{sec:quantitative}


\begin{equation}
\label{eq:std}
\STD = \sqrt{m_2},
\end{equation}

\begin{equation}
\label{eq:skw}
\SKW = \frac{m_3}{m_2^{3/2}},
\end{equation}

\begin{equation}
\label{eq:bi}
\BI = \mathrm{KURTOSIS} - \SKW^2 = \frac{m_4}{m_2^{2}} - \frac{m_3^2}{m_2^{3}},
\end{equation}
